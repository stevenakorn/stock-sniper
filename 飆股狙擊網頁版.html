function renderMoneyFlowData(data, dateStr) { document.getElementById("radarDate").textContent = dateStr; const stockNet = {}; data.forEach(d => { const id = d.stock_id; const buy = parseInt(d.buy) || 0; const sell = parseInt(d.sell) || 0; const net = buy - sell; if (!stockNet[id]) { stockNet[id] = { name: d.name || id, net: 0, foreign: 0, trust: 0, dealer: 0 }; } if (d.name.includes("外資")) stockNet[id].foreign += net; else if (d.name.includes("投信")) stockNet[id].trust += net; else if (d.name.includes("自營商")) stockNet[id].dealer += net; stockNet[id].net += net; }); // 產業分類 const industryFlow = {}; for (const [id, flow] of Object.entries(stockNet)) { const ind = getIndustry(id); if (!industryFlow[ind]) industryFlow[ind] = { foreign:0, trust:0, dealer:0, total:0, stocks:0 }; industryFlow[ind].foreign += flow.foreign; industryFlow[ind].trust += flow.trust; industryFlow[ind].dealer += flow.dealer; industryFlow[ind].total += flow.net; industryFlow[ind].stocks++; } moneyFlowData = Object.entries(industryFlow) .map(([name,f]) => ({name,...f})) .filter(d => d.stocks > 0) .sort((a,b) => b.total - a.total); renderRadar(); // 右側前10 const sorted = Object.values(stockNet) .filter(d => d.net !== 0) .sort((a,b) => b.net - a.net); const top10 = sorted.slice(0,10); const top5s = sorted.filter(d=>d.net<0).slice(-5).reverse(); const maxVal = Math.max(...[...top10,...top5s].map(d=>Math.abs(d.net)))||1; const totalNet = top10.reduce((s,d)=>s+d.net,0); document.getElementById("flowTotal").textContent = (totalNet>=0?"淨買+":"淨賣") + (Math.abs(totalNet)/1000).toFixed(0) + "K張"; document.getElementById("flowChart").innerHTML = `
📅 ${dateStr} 三大法人淨買超前10
` + top10.map(d => { const pct = Math.abs(d.net)/maxVal*88; return `
${esc(d.name)}
+${(d.net/1000).toFixed(1)}K
`; }).join("") + (top5s.length ? `
── 淨賣超 ──
` : "") + top5s.map(d => { const pct = Math.abs(d.net)/maxVal*88; return `
${esc(d.name)}
${(d.net/1000).toFixed(1)}K
`; }).join(""); }
